  const studentRow = Array.isArray(d.users) ? d.users[0] : d.users;

    severity: "info" as const,
    title: studentName,

    subjectName: studentName,
    triggeredAt: d.created_at,
    dismissed: dismissedKeys.has(key),
  };
});
