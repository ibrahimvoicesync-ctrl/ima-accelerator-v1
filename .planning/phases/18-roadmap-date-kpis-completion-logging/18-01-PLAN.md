export function daysInProgram(joinedAt: string): number {
  const joined = new Date(joinedAt);
  joined.setUTCHours(0, 0, 0, 0);
  const now = new Date();
  now.setUTCHours(0, 0, 0, 0);
  return Math.floor((now.getTime() - joined.getTime()) / (1000 * 60 * 60 * 24));
}
