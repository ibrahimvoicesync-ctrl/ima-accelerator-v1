**Reducer function (per D-05):**
```typescript
function dealsReducer(state: Deal[], action: OptimisticAction): Deal[] {
  switch (action.type) {
    case "add":
      return [action.deal, ...state];
    case "edit":
      return state.map((d) => d.id === action.deal.id ? action.deal : d);
    case "delete":
      return state.filter((d) => d.id !== action.id);
  }
}
```
