```typescript
import { NextRequest, NextResponse } from "next/server";

import { ROADMAP_STEPS } from "@/lib/config";
```

**Zod schema:**
```typescript
const patchSchema = z.object({
  step_number: z.number().int().min(1).max(ROADMAP_STEPS.length),
});
```

**PATCH handler flow (exact order):**
1. Auth: `const supabase = await createClient(); const { data: { user: authUser } } = await supabase.auth.getUser();` -- return 401 if no authUser
2. Profile lookup: `const admin = createAdminClient(); const { data: profile } = await admin.from("users").select("id, role").eq("auth_id", authUser.id).single();` -- return 404 if no profile
3. Role check: `if (profile.role !== "student")` -- return 403
4. JSON parse: `try { body = await request.json(); } catch { return 400 "Invalid JSON body" }`
5. Zod validate: `patchSchema.safeParse(body)` -- return 400 with first issue message
6. Fetch step: `admin.from("roadmap_progress").select("*").eq("student_id", profile.id).eq("step_number", step_number).single()` -- return 404 if not found
7. Status check: `if (step.status !== "active")` -- return 400 "Can only complete active steps"
8. Update step: `admin.from("roadmap_progress").update({ status: "completed", completed_at: new Date().toISOString() }).eq("id", step.id).select().single()` -- return 500 on error
9. Unlock next: if `step_number < ROADMAP_STEPS.length`, update `step_number + 1` to `status: "active"` via `admin.from("roadmap_progress").update({ status: "active" }).eq("student_id", profile.id).eq("step_number", step_number + 1).select().single()`
10. Return: `{ data: { completed, unlocked } }`

**Error handling:** Outer try-catch logs `console.error("PATCH /api/roadmap error:", error)` and returns 500.
