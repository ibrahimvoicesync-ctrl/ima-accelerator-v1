  LEFT JOIN work_sessions ws
    ON ws.student_id = u.id

)

-- avg_total_outreach formula (per window):
--   SUM(brands_contacted + influencers_contacted) over reports in window / (COUNT(DISTINCT student_id) * window_days)
-- where window_days: weekly=7, monthly=30, yearly=365,
-- alltime = GREATEST(1, (CURRENT_DATE - MIN(dr.date))::int) computed per coach.

Example CTE:

```sql
coach_outreach_rows AS (
  SELECT
    ec.coach_id, ec.coach_name,
    -- numerator sums (brands + influencers)
    COALESCE(SUM( COALESCE(dr.brands_contacted,0) + COALESCE(dr.influencers_contacted,0) )
      FILTER (WHERE dr.date >= (v_now - INTERVAL '7 days')::date), 0)::numeric AS outreach_sum_weekly,
    COALESCE(SUM( COALESCE(dr.brands_contacted,0) + COALESCE(dr.influencers_contacted,0) )
      FILTER (WHERE dr.date >= (v_now - INTERVAL '30 days')::date), 0)::numeric AS outreach_sum_monthly,
    COALESCE(SUM( COALESCE(dr.brands_contacted,0) + COALESCE(dr.influencers_contacted,0) )
      FILTER (WHERE dr.date >= (v_now - INTERVAL '365 days')::date), 0)::numeric AS outreach_sum_yearly,
    COALESCE(SUM( COALESCE(dr.brands_contacted,0) + COALESCE(dr.influencers_contacted,0) ), 0)::numeric AS outreach_sum_alltime,
    -- denominator: count of distinct students assigned
    COUNT(DISTINCT s.id)::numeric AS student_count,
    -- for alltime window_days
    GREATEST(1, (CURRENT_DATE - MIN(dr.date))::int)::numeric AS alltime_days
  FROM eligible_coaches ec
  LEFT JOIN users s
    ON s.coach_id = ec.coach_id
   AND s.role IN ('student','student_diy')
   AND s.status = 'active'
  LEFT JOIN daily_reports dr ON dr.student_id = s.id
  GROUP BY ec.coach_id, ec.coach_name
),
coach_outreach_avg AS (
  SELECT
    coach_id, coach_name,
    CASE WHEN student_count > 0 THEN outreach_sum_weekly  / (student_count * 7)   ELSE 0 END AS avg_weekly,
    CASE WHEN student_count > 0 THEN outreach_sum_monthly / (student_count * 30)  ELSE 0 END AS avg_monthly,
    CASE WHEN student_count > 0 THEN outreach_sum_yearly  / (student_count * 365) ELSE 0 END AS avg_yearly,
    CASE WHEN student_count > 0 THEN outreach_sum_alltime / (student_count * COALESCE(alltime_days, 1)) ELSE 0 END AS avg_alltime
  FROM coach_outreach_rows
)
```

        'monthly', <hours_monthly_agg>,
        'yearly',  <hours_yearly_agg>,

// Window keys used for every leaderboard (trailing N-days semantics).
export type OwnerAnalyticsWindow = "weekly" | "monthly" | "yearly" | "alltime";

// Shared row fields
type BaseRow = {
  rank: number;
  metric_display: string;
};

// Student rows (unchanged shape per metric, just reorganized under windows).
export type OwnerStudentHoursRow = BaseRow & {
  student_id: string;
  student_name: string;
  minutes: number;
};
export type OwnerStudentProfitRow = BaseRow & {
  student_id: string;
  student_name: string;
  profit: string; // NUMERIC — keep as string from supabase-js
};
export type OwnerStudentDealsRow = BaseRow & {
  student_id: string;
  student_name: string;
  deals: number;
};

// Coach rows (new)
export type OwnerCoachRevenueRow = BaseRow & {
  coach_id: string;
  coach_name: string;
  profit: string; // NUMERIC
};
export type OwnerCoachAvgOutreachRow = BaseRow & {
  coach_id: string;
  coach_name: string;
  avg: string; // NUMERIC — formatted via metric_display
};
export type OwnerCoachDealsRow = BaseRow & {
  coach_id: string;
  coach_name: string;
  deals: number;
};

// Per-window buckets
type Buckets<T> = {
  weekly: T[];
  monthly: T[];
  yearly: T[];
  alltime: T[];
};

export type OwnerLeaderboardsV2 = {
  students: {
    hours: Buckets<OwnerStudentHoursRow>;

    deals: Buckets<OwnerStudentDealsRow>;

    revenue: Buckets<OwnerCoachRevenueRow>;
    avg_total_outreach: Buckets<OwnerCoachAvgOutreachRow>;
    deals: Buckets<OwnerCoachDealsRow>;
  };
};

export type OwnerAnalyticsPayload = {
  leaderboards: OwnerLeaderboardsV2;
};
